from mpython import *
from bluebit import *
import time
import font.digiface_21
import zhrSystem.config as config
from gui import Image

def display_font(_font, _str, _x, _y, _wrap, _z=0):
    _start = _x
    for _c in _str:
        _d = _font.get_ch(_c)
        if _wrap and _x > 128 - _d[2]: _x = _start; _y += _d[1]
        if _c == '1' and _z > 0: oled.fill_rect(_x, _y, _d[2], _d[1], 0)
        oled.blit(framebuf.FrameBuffer(bytearray(_d[0]), _d[2], _d[1],
        framebuf.MONO_HLSB), (_x+int(_d[2]/_z)) if _c=='1' and _z>0 else _x, _y)
        _x += _d[2]

sht20 = SHT20()
start_clock = Clock(oled, 105, 19, 19)
week_day = ['一', '二', '三', '四', '五', '六', '日']
image = Image()

class StartPage():
    def show(self):
        global week_day,image
        while config.get("mode","page")=="start":
            oled.fill(0)
            start_clock.settime()
            display_font(font.digiface_21, (''.join([str(x) for x in [time.localtime()[3] // 10, time.localtime()[3] % 10, ':', time.localtime()[4] // 10, time.localtime()[4] % 10, ':', time.localtime()[5] // 10, time.localtime()[5] % 10]])), 6, 3, False, 2)
            oled.DispChar(str((''.join([str(x) for x in [time.localtime()[1], '月', time.localtime()[2], '日  ', '星期', week_day[(time.localtime()[6]+1 - 1)]]]))), 0, 24, 1)
            oled.DispChar(str((''.join([str(x) for x in ['天气:', config.get('weather','weather'), '  温度:', config.get('weather','low'), '-', config.get('weather','high'), '℃']]))), 0, 37, 1)
            oled.DispChar(str((''.join([str(x) for x in ['当前:', round(sht20.temperature()), '℃  湿度:', round(sht20.humidity()), '%']]))), 0, 50, 1)
            start_clock.drawClock()
            oled.blit(image.load('zhrSystem/Images/have_wifi.bmp', 0), 113, 50)
            oled.show()
